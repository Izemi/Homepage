            def longest_run(sequence, subsequence):
            """Returns the longest run of a subsequence in a DNA sequence."""
            max_run = 0
            subseq_len = len(subsequence)

            for i in range(len(sequence)):
                count = 0
                while sequence[i:i+subseq_len] == subsequence:
                    count += 1
                    i += subseq_len

                max_run = max(max_run, count)

            return max_run
